import java.math.*;
import java.util.*;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		long len = 0;
		String S = "";
		for (int i = 0; i < N; i++) {
			String c = sc.next();
			long l = sc.nextLong();
			len += l;
			if (len > 100) {
				System.out.println("Too Long");
				return;
			}
			S += c.repeat((int) l);
		}
		System.out.println(S);
	}
}
