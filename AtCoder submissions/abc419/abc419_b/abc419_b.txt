import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int q = sc.nextInt();
        Queue<Integer> pq = new PriorityQueue<>();
        for (int qi=0; qi<q; qi++) {
            int x = sc.nextInt();
            if (x == 1) {
                int s = sc.nextInt();
                pq.add(s);
            } else {
                System.out.println(pq.poll());
            }
        }
    }
}
