import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int a = sc.nextInt();
        int b = sc.nextInt();
        String s = sc.next();

        for (int i=0; i<n; i++) {
            if(a<=i && i<n-b) System.out.print(s.charAt(i));
        }
        System.out.println();
    }
}
