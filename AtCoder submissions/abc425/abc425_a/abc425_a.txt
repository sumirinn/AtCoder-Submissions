import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        int ans = 0;
        int cnt = -1;
        for (int i=1; i<=n; i++) {
            ans += cnt*i*i*i;
            cnt *= -1;
        }
        System.out.println(ans);
    }
}
